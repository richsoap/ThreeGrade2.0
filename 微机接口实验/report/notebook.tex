\subsubsection{源码一\_两个十进制数相加}\label{ux6e90ux7801ux4e00ux5fux4e24ux4e2aux5341ux8fdbux5236ux6570ux76f8ux52a0}

\begin{verbatim}
WDTCN DATA 0FFH ;定义地址常量
ORG 0000H ;程序定位
AJMP START ;跳转到标号
ORG 0100H ;程序定位
START:
MOV WDTCN, #0DEH
MOV WDTCN, #0ADH ;禁止看门狗电路复位
MOV R0, #20H ;设置内部数据存储器间接寻址指针
MOV DPTR, #CONST ;设置程序存储器数据指针
CLR A
MOVC A, @A+DPTR ;读程序存储器数据
MOV R1, A ;保存加数低位
CLR A
INC DPTR ;指针地址加1
MOVC A, @A+DPTR
MOV R2, A ;保存加数高位
MOV DPTR, #0000H ;设置外部数据存储器指针1.3 实验内容 5
MOVX A, @DPTR
ADD A, R1 ;先加低位
DA A ;十进制数加法调整
MOV @R0, A ;写内部数据存储器（间接寻址只能@R0或@R1）
INC DPTR
INC R0 ;内部数据指针加1
MOVX A, @DPTR
ADDC A, R2 ;带进位加高位
DA A ;十进制数加法调整
MOV @R0, A ;保存高位
INC R0
CLR A
